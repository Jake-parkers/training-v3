// Using MATCH to retrieve nodes

// tag::simple[]
MATCH (variable)
RETURN variable
// end::simple[]

// tag::label[]
MATCH (variable:Label)
RETURN variable
// end::label[]

// tag::all[]
MATCH (n) 			// returns all nodes in the graph
RETURN n
// end::all[]

// tag::person[]
MATCH (p:Person) 	// returns all Person nodes in the graph
RETURN p
// end::person[]
